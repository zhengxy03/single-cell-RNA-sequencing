    ylab("UMAP_2") +

# 为LUAD和LUSC分别创建组合变量并设置顺序
add_group_response_variable <- function(seurat_obj) {
  # 检查必要的列是否存在
  if (!all(c("LN_group", "pathological_response") %in% colnames(seurat_obj@meta.data))) {
    stop("缺少必要的列: LN_group 或 pathological_response")
  }
  
  # 创建组合变量
  seurat_obj@meta.data$group_response <- paste(seurat_obj@meta.data$LN_group, 
                                               seurat_obj@meta.data$pathological_response, 
                                               sep = "_")
  
  # 定义期望的顺序
  desired_order <- c("LN+_pCR", "LN+_MPR", "LN+_non-MPR", 
                     "LN-_pCR", "LN-_MPR", "LN-_non-MPR")
  
  # 设置因子顺序
  seurat_obj@meta.data$group_response <- factor(seurat_obj@meta.data$group_response,
                                                levels = desired_order)
  
  return(seurat_obj)
}

# 2.1 子函数1：计算细胞数量及占比
calculate_cell_proportions <- function(meta_data, current_group) {
  # 统计每个样本的细胞数
  cell_counts <- meta_data %>%
    filter(LN_group == current_group) %>%
    group_by(patients, cell_type, pathological_response) %>%
    summarise(count = n(), .groups = "drop")
  
  # 每个样本总细胞数
  sample_totals <- cell_counts %>%
    group_by(patients) %>%
    summarise(total = sum(count), .groups = "drop")
  
  # 计算比例并返回
  cell_proportions <- cell_counts %>%
    left_join(sample_totals, by = "patients") %>%
    mutate(proportion = count / total)
  
  return(list(cell_counts = cell_counts, cell_proportions = cell_proportions, sample_totals = sample_totals))
}

# 2.2 子函数2：预处理绘图数据（设置因子、分类、补0）
preprocess_plot_data <- function(cell_proportions, sample_totals) {
  # 设置pathological_response因子水平

    levels = c("pCR", "MPR", "non-MPR")
  )

    pivot_wider(

    )
  
  missing_combinations <- celltype_response_presence %>%
    pivot_longer(-cell_type, names_to = "pathological_response", values_to = "has_data") %>%

  if (nrow(missing_combinations) > 0) {

    )
  )
  
  # 细胞-响应组合的总计数
  cell_response_counts <- cell_counts %>%
    group_by(cell_type, pathological_response) %>%
    summarise(total_count = sum(count), .groups = "drop") %>%
    right_join(full_combinations, by = c("cell_type", "pathological_response")) %>%
    mutate(total_count = replace_na(total_count, 0))
  
  # 各响应类型的全局总计数
  global_response_totals <- cell_counts %>%
    group_by(pathological_response) %>%
    summarise(response_total = sum(count), .groups = "drop") %>%
    right_join(
      data.frame(
        pathological_response = factor(
          c("pCR", "MPR", "non-MPR"),
          levels = c("pCR", "MPR", "non-MPR")
        )
      ),
      by = "pathological_response"
    ) %>%
    mutate(response_total = replace_na(response_total, 0))
  
  return(list(
    cell_response_counts = cell_response_counts,
    global_response_totals = global_response_totals
  ))
}

# 2.4 子函数4：执行卡方检验并整理结果
perform_chisq_test <- function(cell_response_counts, global_response_totals) {
  chisq_results <- cell_response_counts %>%
    group_by(cell_type) %>%
    summarise(
      p_value = {
        obs <- total_count
        exp <- global_response_totals$response_total
        
        if (sum(obs) < 5) {
          NA_real_
        } else {
          tbl <- rbind(obs, exp - obs)
          if (any(tbl < 0)) NA_real_ else chisq.test(tbl)$p.value
        }
      },
      .groups = "drop"
    ) %>%
    mutate(
      significance = case_when(
        is.na(p_value) ~ "ns",
        p_value < 0.001 ~ "***",
        p_value < 0.01 ~ "**",
        p_value < 0.05 ~ "*",
        TRUE ~ "ns"
      ),
      cell_category = classify_cell_type(cell_type)
    )
  
  return(chisq_results)
}

# 2.5 调度函数：简化后的process_group_data（调用各子函数）
process_group_data <- function(current_group) {
  cat("正在处理:", current_group, "\n")

  # 步骤1：计算细胞数量及占比

  cell_counts <- proportion_result$cell_counts
  cell_proportions <- proportion_result$cell_proportions
  sample_totals <- proportion_result$sample_totals

  all_celltypes <- unique(plot_data$cell_type)
  chisq_background <- prepare_chisq_background_data(cell_counts, all_celltypes)
  
  # 步骤4：执行卡方检验
  chisq_results <- perform_chisq_test(
    chisq_background$cell_response_counts,
    chisq_background$global_response_totals
  )
  
  # 返回最终结果
  return(list(plot_data = plot_data, chisq_results = chisq_results))
}

classify_cell_type <- function(cell_type) {
  case_when(
    grepl("^CD4T", cell_type) ~ "CD4 T Cells",
    grepl("^CD8T", cell_type) ~ "CD8 T Cells",
    TRUE ~ "Other T Cells"
  )
}

# 2.1 子函数1：计算细胞数量及占比
calculate_cell_proportions <- function(meta_data, current_group) {
  # 统计每个样本的细胞数
  cell_counts <- meta_data %>%
    filter(LN_group == current_group) %>%
    group_by(patients, cell_type, pathological_response) %>%
    summarise(count = n(), .groups = "drop")
  
  # 每个样本总细胞数
  sample_totals <- cell_counts %>%
    group_by(patients) %>%
    summarise(total = sum(count), .groups = "drop")
  
  # 计算比例并返回
  cell_proportions <- cell_counts %>%
    left_join(sample_totals, by = "patients") %>%
    mutate(proportion = count / total)
  
  return(list(cell_counts = cell_counts, cell_proportions = cell_proportions, sample_totals = sample_totals))
}

# 2.2 子函数2：预处理绘图数据（设置因子、分类、补0）
preprocess_plot_data <- function(cell_proportions, sample_totals) {
  # 设置pathological_response因子水平
  cell_proportions$pathological_response <- factor(
    cell_proportions$pathological_response,
    levels = c("pCR", "MPR", "non-MPR")
  )
  
  # 细胞类型分类
  cell_proportions <- cell_proportions %>%
    mutate(cell_category = classify_cell_type(cell_type))
  
  # 补0操作（保证三类response都有数据）
  celltype_response_presence <- cell_proportions %>%
    group_by(cell_type, pathological_response) %>%
    summarise(has_data = n() > 0, .groups = "drop") %>%
    pivot_wider(
      names_from = pathological_response,
      values_from = has_data,
      values_fill = FALSE
    )
  
  missing_combinations <- celltype_response_presence %>%
    pivot_longer(-cell_type, names_to = "pathological_response", values_to = "has_data") %>%
    filter(!has_data)
  
  if (nrow(missing_combinations) > 0) {
    missing_records <- expand_grid(
      patients = unique(cell_proportions$patients),
      cell_type = missing_combinations$cell_type,
      pathological_response = missing_combinations$pathological_response
    ) %>%
      left_join(sample_totals, by = "patients") %>%
      mutate(
        count = 0,
        proportion = 0,
        cell_category = classify_cell_type(cell_type)
      )
    
    cell_proportions <- bind_rows(cell_proportions, missing_records)
  }
  
  return(cell_proportions)
}

# 2.4 子函数4：执行检验（卡方 + Fisher精确检验 自适应切换）
perform_stat_test <- function(cell_response_counts, global_response_totals) {
  stat_results <- cell_response_counts %>%
    group_by(cell_type) %>%
    summarise(
      p_value = {
        obs <- total_count
        exp <- global_response_totals$response_total
        
        # 构建列联表（细胞类型的各响应计数 vs 全局其他细胞的各响应计数）
        tbl <- rbind(obs, exp - obs)
        # 过滤全0行/列（避免检验报错）
        tbl <- tbl[rowSums(tbl) > 0, colSums(tbl) > 0, drop = FALSE]
        
        if (nrow(tbl) < 2 | ncol(tbl) < 2) {
          # 行列数不足，无法检验
          NA_real_
        } else if (any(tbl < 5) | sum(obs) < 10) {
          # 小样本/低频数：用Fisher精确检验
          fisher.test(tbl)$p.value
        } else {
          # 大样本：用卡方检验
          chisq.test(tbl)$p.value
        }
      },
      .groups = "drop"
    ) %>%
    mutate(
      significance = case_when(
        is.na(p_value) ~ "ns",
        p_value < 0.001 ~ "***",
        p_value < 0.01 ~ "**",
        p_value < 0.05 ~ "*",
        TRUE ~ "ns"
      ),
      cell_category = classify_cell_type(cell_type)
    )
  
  return(stat_results)
}
